#include <iostream>

using namespace std;

int F[1001][1001] = { 0 }; //F[i][j] : so cach duoc tong =j mà không có i
//	F[i][j] = F[i-1][j]+ F[i][j-i]
void chiaNhom()
{
    int numTest;
    cin >> numTest;
    //dynamic

    for (int i = 0; i <= 1000; i++) {
        F[i][0] = 1;
    }
    for (int i = 2; i <= 1000; i++) {
        for (int j = 2; j <= 1000; j++) {
            int x = (j >= i) ? F[i][j - i] : 0;
            F[i][j] = x + F[i - 1][j];
            F[i][j] = F[i][j] % 1000000007;
            //cout<<F[i][j]<<" ";
        }
        //cout<<"\n";
    }

    for (int tCase = 0; tCase < numTest; tCase++) {
        //input
        int n;
        cin >> n;
        //show output
        cout << F[n][n] << "\n";
    }
}

int main()
{

    chiaNhom();

    return 0;
}
