	void cmp(){
		for(int i=1; i<=15; i++){

	void backtrack(int x, int y,int row,int col){
		if(x==row && y==col){
			if(res<matrix[row][col]){
				res=matrix[row][col];
				cmp();
			}
		}
		if(x<row){
			temp [x+1][y]=1;
			matrix[x+1][y]+=matrix[x][y];
			backtrack(x+1 , y , row, col);
			matrix[x+1][y]-=matrix[x][y];
			temp [x+1][y]=0;
		}
		if(y<col){
			temp [x][y+1]=1;
			matrix[x][y+1]+=matrix[x][y];
			backtrack( x, y+1 , row, col);
			matrix[x][y+1]-=matrix[x][y];
			temp [x][y+1]=0;
		}
	}
